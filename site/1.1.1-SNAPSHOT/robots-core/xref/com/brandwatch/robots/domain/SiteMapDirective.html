36  import com.google.common.base.MoreObjects;
37  import com.google.common.base.Objects;
38  
39  import javax.annotation.Nonnull;
40  import javax.annotation.Nullable;
41  import javax.annotation.concurrent.Immutable;
42  
43  import static com.google.common.base.Preconditions.checkNotNull;
44  
45  @Immutable
46  public final class SiteMapDirective implements Directive {
47  
48      @Nonnull
49      private final String value;
50  
51      public SiteMapDirective(@Nonnull String value) {
52          this.value = checkNotNull(value, "value is null");
53      }
54  
55      @Override
56      @Nonnull
57      public String getField() {
58          return "sitemap";
59      }
60  
61      @Override
62      @Nonnull
63      public String getValue() {
64          return value;
65      }
66  
67      @Override
68      public boolean equals(@Nullable Object o) {
69          if (this == o) return true;
70          if (o == null || getClass() != o.getClass()) return false;
71          SiteMapDirective that = (SiteMapDirective) o;
72          return Objects.equal(value, that.value);
73      }
74  
75      @Override
76      public int hashCode() {
77          return Objects.hashCode(value);
78      }
79  
80      @Nonnull
81      @Override
82      public String toString() {
83          return MoreObjects.toStringHelper(this)
84                  .add("value", value)
85                  .toString();
86      }
87  }
