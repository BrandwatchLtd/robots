36  import com.brandwatch.robots.domain.Robots;
37  import com.brandwatch.robots.net.CharSourceSupplier;
38  import com.brandwatch.robots.net.LoggingReader;
39  import com.brandwatch.robots.parser.ParseException;
40  import com.brandwatch.robots.parser.RobotsParser;
41  import com.brandwatch.robots.util.LogLevel;
42  import com.google.common.io.CharSource;
43  import com.google.common.io.Closer;
44  import org.slf4j.Logger;
45  import org.slf4j.LoggerFactory;
46  
47  import javax.annotation.Nonnull;
48  import java.io.IOException;
49  import java.io.Reader;
50  import java.net.URI;
51  import java.net.URISyntaxException;
52  import java.util.concurrent.TimeoutException;
53  
54  import static com.google.common.base.Preconditions.checkNotNull;
55  import static java.text.MessageFormat.format;
56  
57  final class RobotsLoaderImpl implements RobotsLoader {
58  
59      private static final Logger log = LoggerFactory.getLogger(RobotsLoaderImpl.class);
60      @Nonnull
61      private final RobotsFactory factory;
62      @Nonnull
63      private final CharSourceSupplier charSourceSupplier;
64  
65      public RobotsLoaderImpl(@Nonnull RobotsFactory factory) {
66          this.factory = checkNotNull(factory, "factory is null");
67          this.charSourceSupplier = checkNotNull(factory.createCharSourceSupplier());
68      }
69  
70      @Nonnull
71      @Override
72      public Robots load(@Nonnull URI robotsResource) {
73          checkNotNull(robotsResource, "robotsResource");
74          log.debug("Loading: {}", robotsResource);
75          return load(robotsResource, charSourceSupplier.get(robotsResource));
76      }
77  
78      public void close() throws IOException {
79          charSourceSupplier.close();
80      }
81  
82      @Nonnull
83      private Robots load(@Nonnull URI robotsResource, @Nonnull CharSource robotsData) {
84          try {
85              final Closer closer = Closer.create();
86              try {
87  
88                  final Reader reader;
89                  try {
90                      reader = closer.register(robotsData.openBufferedStream());
91                  } catch (TemporaryAllow e) {
92                      return fullAllow(robotsResource, e.getMessage());
93                  } catch (TemporaryDisallow e) {
94                      return fullDisallow(robotsResource, e.getMessage());
95                  }
96                  return conditionalAllow(robotsResource, reader);
97              } catch (Throwable t) {
98                  throw closer.rethrow(t);
99              } finally {
100                 closer.close();
101             }
102         } catch (IOException e) {
103             if(e.getCause() instanceof TimeoutException) {
104                 return fullAllow(robotsResource, format("Timeout waiting for response."));
105             } else {
106                 log.debug("Caught IO exception", e);
107                 return fullAllow(robotsResource, format("IO exception: \"{0}\"", e.getMessage()));
108             }
109         }
110     }
111 
112     @Nonnull
113     private Robots conditionalAllow(@Nonnull URI robotsResource, @Nonnull Reader robotsData)
114             throws IOException {
115         log.debug("Conditional allow; parsing contents of {}", robotsResource);
116 
117         final Reader reader = new LoggingReader(robotsData, this.getClass(), LogLevel.TRACE);
118         final RobotsParser parser = new RobotsParser(reader);
119         final RobotsBuildingParseHandler handler = factory.createRobotsBuildingHandler();
120 
121         try {
122             parser.parse(handler);
123             return handler.get();
124         } catch (ParseException e) {
125             return fullAllow(robotsResource, format("Caught parsing exception: \"{0}\"", e));
126         }
127     }
128 
129     @Nonnull
130     private Robots fullAllow(@Nonnull URI robotsResource, @Nonnull String reason, @Nonnull Object... args) {
131         log.info("Allowing entire site: {}; {}", site(robotsResource), format(reason, args));
132         return factory.createAllowAllRobots();
133     }
134 
135     @Nonnull
136     private Robots fullDisallow(@Nonnull URI robotsResource, @Nonnull String reason, @Nonnull Object... args) {
137         log.info("Disallowing entire site: {}; {}", site(robotsResource), format(reason, args));
138         return factory.createDisallowAllRobots();
139     }
140 
141     static URI site(URI resource) {
142         try {
143             return new URI(resource.getScheme(), resource.getUserInfo(), resource.getHost(),
144                     resource.getPort(), null, null, null);
145         } catch (URISyntaxException e) {
146             throw new AssertionError(e);
147         }
148     }
