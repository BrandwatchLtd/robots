36  import com.google.common.base.Function;
37  
38  import javax.annotation.Nonnull;
39  import javax.annotation.Nullable;
40  
41  import static com.google.common.base.Preconditions.checkNotNull;
42  
43  class MissingPrefixFixingFunction implements Function<String, String> {
44  
45      @Nullable
46      @Override
47      public String apply(@Nonnull String input) {
48          checkNotNull(input, "input is null");
49          if (!input.isEmpty()) {
50              final char c = input.charAt(0);
51              if (c != '/' && c != '*' && c != '^') {
52                  return '/' + input;
53              }
54          }
55          return input;
56      }
57  }
