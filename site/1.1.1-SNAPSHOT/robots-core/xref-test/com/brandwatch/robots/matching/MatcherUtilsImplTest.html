36  import com.google.common.base.Optional;
37  import com.google.common.collect.ImmutableList;
38  import org.junit.Before;
39  import org.junit.Test;
40  
41  import java.util.Collections;
42  import java.util.List;
43  
44  import static org.hamcrest.Matchers.equalTo;
45  import static org.junit.Assert.assertThat;
46  
47  public class MatcherUtilsImplTest {
48  
49      private static ExpressionCompiler agentExpressionCompiler = new ExpressionCompilerBuilder()
50              .withCaseSensitivity(false)
51              .build();
52  
53      protected MatcherUtilsImpl utilities;
54  
55      private static Matcher<String> newPatternMatcher(String pattern) {
56          return agentExpressionCompiler.compile(pattern);
57      }
58  
59      @Before
60      public final void setup() {
61          utilities = new MatcherUtilsImpl();
62      }
63  
64      @Test(expected = NullPointerException.class)
65      public void givenMatcherIsNull_whenGetMatchSpecificity_thenThrowsNPE() {
66          Matcher<String> matcher = null;
67          String agent = "googlebot";
68          utilities.getMatchSpecificity(matcher, agent);
69      }
70  
71      @Test(expected = NullPointerException.class)
72      public void givenAgentIsNull_whenGetMatchSpecificity_thenThrowsNPE() {
73          Matcher<String> matcher = newPatternMatcher("*");
74          String agent = null;
75          utilities.getMatchSpecificity(matcher, agent);
76      }
77  
78      @Test
79      public void givenPatternIsWildcard_whenGetMatchSpecificity_thenReturnsZero() {
80          Matcher<String> matcher = newPatternMatcher("*");
81          String agent = "googlebot";
82          Optional<Double> specificity = utilities.getMatchSpecificity(matcher, agent);
83          assertThat(specificity.get(), equalTo(0.0));
84      }
85  
86      @Test
87      public void givenPatternMatches_whenGetMatchSpecificity_thenReturnsMatcherSpecificity() {
88          Matcher<String> matcher = newPatternMatcher("google");
89          String agent = "googlebot";
90          Optional<Double> specificity = utilities.getMatchSpecificity(matcher, agent);
91          assertThat(specificity.get(), equalTo(matcher.getSpecificity()));
92      }
93  
94      @Test
95      public void givenPatternMatchesIgnoringCase_whenGetMatchSpecificity_thenReturnsMatcherSpecificity() {
96          Matcher<String> matcher = newPatternMatcher("gOoGlE");
97          String agent = "googlebot";
98          Optional<Double> specificity = utilities.getMatchSpecificity(matcher, agent);
99          assertThat(specificity.get(), equalTo(matcher.getSpecificity()));
100     }
101 
102     @Test
103     public void givenPatternNotMatches_whenGetMatchSpecificity_thenReturnsAbsent() {
104         Matcher<String> matcher = newPatternMatcher("yahoo");
105         String agent = "googlebot";
106         Optional<Double> specificity = utilities.getMatchSpecificity(matcher, agent);
107         assertThat(specificity, equalTo(Optional.<Double>absent()));
108     }
109 
110     @Test
111     public void givenPatternIsEmpty_whenGetMatchSpecificity_thenReturnsZero() {
112         Matcher<String> matcher = newPatternMatcher("");
113         String agent = "googlebot";
114         Optional<Double> specificity = utilities.getMatchSpecificity(matcher, agent);
115         assertThat(specificity.get(), equalTo(0.0));
116     }
117 
118     @Test
119     public void givenPatternAndAgentAreEmpty_whenGetMatchSpecificity_thenReturnsZero() {
120         Matcher<String> matcher = newPatternMatcher("");
121         String agent = "";
122         Optional<Double> specificity = utilities.getMatchSpecificity(matcher, agent);
123         assertThat(specificity.get(), equalTo(0.0));
124     }
125 
126     @Test
127     public void givenWildcardPattern_andAgentIsEmpty_whenGetMatchSpecificity_thenReturnsZero() {
128         Matcher<String> matcher = newPatternMatcher("*");
129         String agent = "";
130         Optional<Double> specificity = utilities.getMatchSpecificity(matcher, agent);
131         assertThat(specificity.get(), equalTo(0.0));
132     }

139     }

144         String agent = null;
145         utilities.getMostSpecificMatchSpecificity(group, agent);
146     }

148     @Test
149     public void givenGroupIsEmpty_whenGetMostSpecificMatchSpecificity_thenReturnsAbsent() {
150         List<Matchable<String>> group = ImmutableList.of();
151         String agent = "googlebot";
152         Optional<Double> longestMatch = utilities.getMostSpecificMatchSpecificity(group, agent);
153         assertThat(longestMatch, equalTo(Optional.<Double>absent()));
154     }
155 
156     @Test
157     public void givenSingltonWildcardGroup_whenGetMostSpecificMatchSpecificity_thenReturnsZero() {
158         List<Matchable<String>> group = ImmutableList
159                 .<Matchable<String>>builder()
160                 .add(new Matchable<String>() {
161                     @Override
162                     public Matcher<String> getMatcher() {
163                         return newPatternMatcher("*");
164                     }
165                 })
166                 .build();
167         String agent = "googlebot";
168         Optional<Double> longestMatch = utilities.getMostSpecificMatchSpecificity(group, agent);
169         assertThat(longestMatch.get(), equalTo(0.0));
170     }

178     }
179 
180     @Test(expected = NullPointerException.class)
181     public void givenAgentIsNull_whenGetMostSpecificMatchingGroup_thenThrowsNPE() {

184         utilities.getMostSpecificMatchingGroup(groups, agentString);
185     }
186 
187     @Test
188     public void givenGroupsAreEmpty_whenGetMostSpecificMatchingGroup_thenReturnsAbsent() {
189         List<List<Matchable<String>>> groups = Collections.emptyList();
190         String agentString = "googlebot";
191         Optional<List<Matchable<String>>> result = utilities.getMostSpecificMatchingGroup(groups, agentString);
192         assertThat(result, equalTo(Optional.<List<Matchable<String>>>absent()));
193     }
