36  import com.brandwatch.robots.domain.AgentDirective;
37  import com.brandwatch.robots.domain.Group;
38  import com.google.common.base.Optional;
39  import com.google.common.collect.ImmutableList;
40  import org.junit.Before;
41  import org.junit.Test;
42  import org.junit.runner.RunWith;
43  import org.junit.runners.Parameterized;
44  
45  import javax.annotation.Nonnull;
46  import java.util.List;
47  
48  import static org.hamcrest.Matchers.equalTo;
49  import static org.junit.Assert.assertThat;
50  
51  @RunWith(Parameterized.class)
52  public class MatcherUtilsGetBestMatchingGroupDataTests {
53  
54  
55      private static ExpressionCompiler agentExpressionCompiler = new ExpressionCompilerBuilder()
56              .withCaseSensitivity(false)
57              .build();
58  
59      protected MatcherUtilsImpl utilities;
60  
61      @Before
62      public final void setup() {
63          utilities = new MatcherUtilsImpl();
64      }
65  
66      @Nonnull
67      private static final Group firstGroup = new Group.Builder()
68              .withDirective(new AgentDirective("googlebot-news", agentExpressionCompiler.compile("googlebot-news")))
69              .build();
70      @Nonnull
71      private static final Group secondGroup = new Group.Builder()
72              .withDirective(new AgentDirective("*", agentExpressionCompiler.compile("*")))
73              .build();
74      @Nonnull
75      private static final Group thirdGroup = new Group.Builder()
76              .withDirective(new AgentDirective("googlebot", agentExpressionCompiler.compile("googlebot")))
77              .build();
78      private static final List<Group> groups = ImmutableList.<Group>builder()
79              .add(firstGroup).add(secondGroup).add(thirdGroup)
80              .build();
81  
82  
83      private final String crawlerName;
84      private final Group expectedGroup;
85  
86      public MatcherUtilsGetBestMatchingGroupDataTests(String crawlerName, Group expectedGroup) {
87          this.crawlerName = crawlerName;
88          this.expectedGroup = expectedGroup;
89      }
90  
91      @Parameterized.Parameters()
92      public static Iterable<Object[]> data() {
93          return ImmutableList.<Object[]>builder()
94                  .add(new Object[]{"Googlebot", thirdGroup})
95                  .add(new Object[]{"Googlebot-News", firstGroup})
96                  .add(new Object[]{"Googlebot-web", thirdGroup})
97                  .add(new Object[]{"Googlebot-Images", thirdGroup})
98                  .add(new Object[]{"Googlebot-News", firstGroup})
99                  .add(new Object[]{"Otherbot-News", secondGroup})
100                 .add(new Object[]{"Otherbot-Web", secondGroup})
101                 .build();
102     }
103 
104     @Test
105     public void whenGetBestMatchingGroup_thenResultEqualsExpected() {
106         Optional<Group> bestMatch = utilities.getMostSpecificMatchingGroup(groups, crawlerName);
107         assertThat(bestMatch.get(), equalTo(expectedGroup));
108     }
109 
110     @Test
111     public void givenUpperCaseCrawlerName_whenGetBestMatchingGroup_thenResultEqualsExpected() {
112         Optional<Group> bestMatch = utilities.getMostSpecificMatchingGroup(groups, crawlerName.toUpperCase());
113         assertThat(bestMatch.get(), equalTo(expectedGroup));
114     }
115 
116     @Test
117     public void givenLowerCaseCrawlerName_whenGetBestMatchingGroup_thenResultEqualsExpected() {
118         Optional<Group> bestMatch = utilities.getMostSpecificMatchingGroup(groups, crawlerName.toLowerCase());
119         assertThat(bestMatch.get(), equalTo(expectedGroup));
120     }
121 
122 }
