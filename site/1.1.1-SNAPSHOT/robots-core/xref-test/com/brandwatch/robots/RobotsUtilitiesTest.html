36  import org.junit.Before;
37  import org.junit.Test;
38  
39  import java.net.URI;
40  
41  import static org.hamcrest.Matchers.equalTo;
42  import static org.junit.Assert.assertThat;
43  
44  public class RobotsUtilitiesTest {
45  
46      private RobotsUtilities utilities;
47  
48      @Before
49      public void setup() {
50          utilities = new RobotsUtilities();
51      }
52  
53      @Test(expected = NullPointerException.class)
54      public void givenNullResource_whenGetResourceLocalComponents_thenThrowsNPE() {
55          URI resource = null;
56          utilities.getResourceLocalComponents(resource);
57      }
58  
59      @Test
60      public void givenPath_whenGetResourceLocalComponents_thenResultEqualsPath() {
61          URI resource = URI.create("http://example.com/path");
62          String path = utilities.getResourceLocalComponents(resource);
63          assertThat(path, equalTo("/path"));
64      }
65  
66      @Test
67      public void givenFragment_whenGetResourceLocalComponents_thenResultEqualsFragment() {
68          URI resource = URI.create("http://example.com#fragment");
69          String result = utilities.getResourceLocalComponents(resource);
70          assertThat(result, equalTo("#fragment"));
71      }
72  
73      @Test
74      public void givenQuery_whenGetResourceLocalComponents_thenResultEqualsQuery() {
75          URI resource = URI.create("http://example.com?query");
76          String result = utilities.getResourceLocalComponents(resource);
77          assertThat(result, equalTo("?query"));
78      }
79  
80      @Test
81      public void givenPathAndQuery_whenGetResourceLocalComponents_thenResultEqualsExpected() {
82          URI resource = URI.create("http://example.com/path?query");
83          String result = utilities.getResourceLocalComponents(resource);
84          assertThat(result, equalTo("/path?query"));
85      }
86  
87      @Test
88      public void givenPathAndFragment_whenGetResourceLocalComponents_thenResultEqualsExpected() {
89          URI resource = URI.create("http://example.com/path#fragment");
90          String result = utilities.getResourceLocalComponents(resource);
91          assertThat(result, equalTo("/path#fragment"));
92      }
93  
94      @Test
95      public void givenPathAndQueryAndFragment_whenGetResourceLocalComponents_thenResultEqualsExpected() {
96          URI resource = URI.create("http://example.com/path?query#fragment");
97          String result = utilities.getResourceLocalComponents(resource);
98          assertThat(result, equalTo("/path?query#fragment"));
99      }
100 
101     @Test(expected = IllegalArgumentException.class)
102     public void givenMalformedResourceURI_whenGetRobotsURIForResource_thenThrowsIAE() {
103         URI resourceUri = URI.create("http://mobil..bloggplatsen.se/rss/");
104         utilities.getRobotsURIForResource(resourceUri);
105     }
106 
107     @Test(expected = IllegalArgumentException.class)
108     public void givenAnotherMalformedResourceURI_whenGetRobotsURIForResource_thenThrowsIAE() {
109         URI resourceUri = URI.create("http://.mattbrailsford.com/2010/07/15/10-essential-umbraco-packages-for-seo/");
110         utilities.getRobotsURIForResource(resourceUri);
111     }
112 }
