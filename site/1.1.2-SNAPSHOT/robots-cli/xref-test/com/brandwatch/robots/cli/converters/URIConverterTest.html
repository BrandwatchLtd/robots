36  import com.beust.jcommander.ParameterException;
37  import org.junit.Before;
38  import org.junit.Test;
39  
40  import java.net.URI;
41  
42  import static org.hamcrest.Matchers.equalTo;
43  import static org.junit.Assert.assertThat;
44  
45  public class URIConverterTest {
46  
47      private URIConverter converter;
48  
49      @Before
50      public void setup() {
51          converter = new URIConverter();
52      }
53  
54      @Test(expected = NullPointerException.class)
55      public void givenValueIsNull_whenConvert_thenThrowsNullPointerException() {
56          converter.convert(null);
57      }
58  
59      @Test
60      public void givenBlankURI_whenConvert_thenReturnsExpectedURI() {
61          URI result = converter.convert("");
62          assertThat(result, equalTo(URI.create("")));
63      }
64  
65      @Test(expected = ParameterException.class)
66      public void givenInvalidURI_whenConvert_thenThrowsParameterException() {
67          converter.convert(":");
68      }
69  
70      @Test
71      public void givenValidURI_whenConvert_thenReturnsExpectedURI() {
72          URI result = converter.convert("http://www.example.com/index.html");
73          assertThat(result, equalTo(URI.create("http://www.example.com/index.html")));
74      }
75  
76      @Test
77      public void givenValidIDN_whenConvert_thenReturnsExpectedURI() {
78          URI result = converter.convert("http://detrèsbonsdomaines.com");
79          assertThat(result, equalTo(URI.create("http://xn--detrsbonsdomaines-vsb.com")));
80      }
81  
82      @Test
83      public void givenValidIDN2_whenConvert_thenReturnsExpectedURI() {
84          URI result = converter.convert("http://名がドメイン.com");
85          assertThat(result, equalTo(URI.create("http://xn--v8jxj3d1dzdz08w.com")));
86      }
87  
88  
89      @Test
90      public void givenValidIDN3_whenConvert_thenReturnsExpectedURI() {
91          URI result = converter.convert("http://detrèsbonsdomaines.com");
92          assertThat(result.toString(), equalTo("http://xn--detrsbonsdomaines-vsb.com"));
93      }
94  
95  
96  }
