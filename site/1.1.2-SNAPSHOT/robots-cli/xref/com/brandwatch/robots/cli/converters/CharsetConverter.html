36  import com.beust.jcommander.IStringConverter;
37  import com.beust.jcommander.ParameterException;
38  
39  import java.nio.charset.Charset;
40  import java.nio.charset.IllegalCharsetNameException;
41  import java.nio.charset.UnsupportedCharsetException;
42  
43  import static com.google.common.base.Preconditions.checkNotNull;
44  
45  public class CharsetConverter implements IStringConverter<Charset> {
46  
47      @Override
48      public Charset convert(String parameterValue) {
49          checkNotNull(parameterValue, "parameterValue is null");
50          try {
51              return Charset.forName(parameterValue);
52          } catch (UnsupportedCharsetException e) {
53              throw new ParameterException("Unsupported character set: " + e.getMessage());
54          } catch (IllegalCharsetNameException e) {
55              throw new ParameterException("Illegal character set name: " + e.getMessage());
56          }
57      }
58  
59  }
