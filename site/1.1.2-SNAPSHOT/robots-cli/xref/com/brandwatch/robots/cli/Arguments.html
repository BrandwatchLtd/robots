36  import com.beust.jcommander.Parameter;
37  import com.beust.jcommander.validators.PositiveInteger;
38  import com.brandwatch.robots.RobotsConfig;
39  import com.brandwatch.robots.cli.converters.CharsetConverter;
40  import com.brandwatch.robots.cli.converters.URIConverter;
41  import com.brandwatch.robots.cli.validators.AbsoluteURIValidator;
42  import com.brandwatch.robots.cli.validators.NonEmptyStringValidator;
43  import com.google.common.base.Charsets;
44  
45  import javax.annotation.Nonnegative;
46  import javax.annotation.Nonnull;
47  import java.net.URI;
48  import java.nio.charset.Charset;
49  import java.util.List;
50  
51  import static com.google.common.collect.Lists.newArrayList;
52  
53  public class Arguments {
54  
55      @Parameter(
56              names = {"--help", "-h"},
57              description = "Display this helpful message.",
58              help = true
59      )
60      private boolean helpRequested = false;
61  
62      @Nonnegative
63      @Parameter(
64              names = {"--maxFileSizeBytes", "-s"},
65              description = "Download size limit. robots.txt retrieval will give up beyond this point.",
66              validateWith = PositiveInteger.class
67      )
68      private int maxFileSizeBytes = 192 * 1024;
69  
70      @Nonnegative
71      @Parameter(
72              names = {"--maxRedirectHops", "-r"},
73              description = "Number of HTTP 3XX (redirection) responses to follow before giving up.",
74              validateWith = PositiveInteger.class
75      )
76      private int maxRedirectHops = 5;
77  
78      @Nonnull
79      @Parameter(
80              names = {"--defaultCharset", "--charset", "-c"},
81              description = "Preferred character encoding for reading robots.txt. Used when server doesn't specify encoding.",
82              converter = CharsetConverter.class
83      )
84      private Charset defaultCharset = Charsets.UTF_8;
85  
86      @Nonnull
87      @Parameter(
88              names = {"--agent", "-a"},
89              description = "User agent identifier. Sent to the host on retrieval of robots.txt, and also used for directive group matching.",
90              validateWith = NonEmptyStringValidator.class
91      )
92      private String agent = "<unnamed-agent>";
93  
94      @Nonnull
95      @Parameter(
96              required = true,
97              description = "RESOURCES",
98              converter = URIConverter.class,
99              validateWith = AbsoluteURIValidator.class
100     )
101     private List<URI> resources = newArrayList();
102 
103     @Nonnegative
104     @Parameter(
105             names = {"--readTimeout", "-t"},
106             description = "Time in millis before the client times out while downloading a response payload.",
107             validateWith = PositiveInteger.class
108     )
109     private int readTimeoutMillis = 30000;
110 
111     @Nonnegative
112     public int getMaxFileSizeBytes() {
113         return maxFileSizeBytes;
114     }
115 
116     @Nonnegative
117     public int getMaxRedirectHops() {
118         return maxRedirectHops;
119     }
120 
121     @Nonnull
122     public Charset getDefaultCharset() {
123         return defaultCharset;
124     }
125 
126     @Nonnull
127     public List<URI> getResources() {
128         return resources;
129     }
130 
131     @Nonnull
132     public String getAgent() {
133         return agent;
134     }
135 
136     public boolean isHelpRequested() {
137         return helpRequested;
138     }
139 
140     @Nonnegative
141     public int getReadTimeoutMillis() {
142         return readTimeoutMillis;
143     }
144 
145     @Nonnull
146     public RobotsConfig buildRobotsConfig() {
147         final RobotsConfig config = new RobotsConfig();
148         config.setMaxFileSizeBytes(getMaxFileSizeBytes());
149         config.setMaxRedirectHops(getMaxRedirectHops());
150         config.setDefaultCharset(getDefaultCharset());
151         config.setReadTimeoutMillis(getReadTimeoutMillis());
152         return config;
153     }
154 
155 }
