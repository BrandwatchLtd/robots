37  import com.brandwatch.robots.parser.RobotsParserImpl;
38  import com.google.common.base.Charsets;
39  import com.google.common.collect.ImmutableList;
40  import com.google.common.io.Resources;
41  import org.junit.Before;
42  import org.junit.runners.Parameterized;
43  
44  import javax.annotation.Nonnull;
45  import java.io.IOException;
46  import java.io.Reader;
47  
48  import static com.google.common.io.Resources.getResource;
49  
50  public abstract class AbstractDataTest {
51  
52      private final String resourceName;
53      protected RobotsParser robotsTxtParser;
54  
55      protected AbstractDataTest(String resourceName) {
56          this.resourceName = resourceName;
57      }
58  
59      @Parameterized.Parameters(name = "{index}: {0}")
60      public static Iterable<Object[]> data() {
61          return ImmutableList.<Object[]>builder()
62                  .add(new Object[]{"http_boards.dailymail.co.uk_robots.txt"})
63                  .add(new Object[]{"https_www.drupal.org_robots.txt"})
64                  .add(new Object[]{"http_www.brandwatch.com_robots.txt"})
65                  .add(new Object[]{"http_www.dailymail.co.uk_robots.txt"})
66                  .add(new Object[]{"http_www.google.com_robots.txt"})
67                  .add(new Object[]{"https_twitter.com_robots.txt"})
68                  .add(new Object[]{"http_facebook.com_robots.txt"})
69                  .add(new Object[]{"obama_http_www.whitehouse.gov_robots.txt"})
70                  .add(new Object[]{"bush_http_www.whitehouse.gov_robots.txt"})
71                  .add(new Object[]{"http_www.last.fm_robots.txt"})
72                  .add(new Object[]{"http_www.reddit.com_robots.txt"})
73                  .build();
74      }
75  
76      public static Reader resourceReader(@Nonnull String name) throws IOException {
77          return Resources.asCharSource(
78                  getResource(AbstractDataTest.class, name), Charsets.UTF_8
79          ).openBufferedStream();
80      }
81  
82      @Before
83      public final void setupAbstractTest() throws IOException {
84          robotsTxtParser = new RobotsParserImpl(resourceReader(resourceName));
85      }
86  
87  }
