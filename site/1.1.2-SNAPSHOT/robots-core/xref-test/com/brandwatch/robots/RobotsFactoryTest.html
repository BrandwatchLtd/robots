108     private void assertRobotsIsValidSingletonWithGivenFieldType(Robots robots, PathDirective.Field expectedField) {
109         assertThat(robots, notNullValue());
110 
111         assertThat(robots.getGroups(), hasSize(1));
112         Group group = getOnlyElement(robots.getGroups());
113         assertThat(group, Matchers.notNullValue());
114 
115         assertThat(group.getDirectives(AgentDirective.class), hasSize(1));
116         AgentDirective agentDirective = getOnlyElement(group.getDirectives(AgentDirective.class));
117         assertThat(agentDirective, Matchers.notNullValue());
118         assertTrue(agentDirective.getMatcher().matches("abc123"));
119 
120         assertThat(group.getDirectives(PathDirective.class), hasSize(1));
121         PathDirective pathDirective = getOnlyElement(group.getDirectives(PathDirective.class));
122         assertThat(pathDirective, Matchers.notNullValue());
123         assertTrue(pathDirective.getMatcher().matches("abc123"));
124         assertThat(pathDirective.getFieldValue(), equalTo(expectedField));
125     }
