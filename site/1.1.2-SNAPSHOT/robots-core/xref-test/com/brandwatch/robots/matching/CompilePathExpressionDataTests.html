158     @Test
159     public void whenIsPathPatternMatched_thenResultEqualsExpected() {
160         Matcher<String> pattern = pathExpressionCompiler.compile(pathPattern);
161         boolean result = pattern.matches(uri.getPath());
162         assertThat(result, equalTo(expectedResult));
163     }
