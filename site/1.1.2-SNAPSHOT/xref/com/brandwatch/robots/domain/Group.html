36  import com.google.common.base.MoreObjects;
37  import com.google.common.base.Objects;
38  import com.google.common.collect.ImmutableList;
39  
40  import javax.annotation.Nonnull;
41  import javax.annotation.Nullable;
42  import javax.annotation.concurrent.Immutable;
43  import java.util.Iterator;
44  import java.util.List;
45  
46  import static com.google.common.base.Preconditions.checkNotNull;
47  
48  @Immutable
49  public final class Group implements Iterable<AgentDirective> {
50  
51      @Nonnull
52      private final List<Directive> directives;
53  
54      public Group(@Nonnull Builder builder) {
55          this.directives = builder.directives.build();
56      }
57  
58      @Nonnull
59      public List<Directive> getDirectives() {
60          return directives;
61      }
62  
63      @Nonnull
64      public <T extends Directive> List<T> getDirectives(@Nonnull Class<T> directiveType) {
65          ImmutableList.Builder<T> result = ImmutableList.builder();
66          for (Directive directive : directives) {
67              if (directiveType.isAssignableFrom(directive.getClass())) {
68                  result.add(directiveType.cast(directive));
69              }
70          }
71          return result.build();
72      }
73  
74      @Override
75      public boolean equals(@Nullable Object o) {
76          if (this == o) return true;
77          if (o == null || getClass() != o.getClass()) return false;
78          Group group = (Group) o;
79          return Objects.equal(directives, group.directives);
80      }
81  
82      @Override
83      public int hashCode() {
84          return Objects.hashCode(directives);
85      }
86  
87      @Nonnull
88      @Override
89      public String toString() {
90          return MoreObjects.toStringHelper(this)
91                  .add("directives", directives)
92                  .toString();
93      }
94  
95      @Override
96      public Iterator<AgentDirective> iterator() {
97          List<AgentDirective> agentDirectives = getDirectives(AgentDirective.class);
98          return agentDirectives.iterator();
99      }
100 
101     public static class Builder {
102 
103         @Nonnull
104         private final ImmutableList.Builder<Directive> directives = ImmutableList.builder();
105 
106         @Nonnull
107         public Builder withDirective(@Nonnull Directive directive) {
108             directives.add(checkNotNull(directive, "directive is null"));
109             return this;
110         }
111 
112         @Nonnull
113         public Group build() {
114             return new Group(this);
115         }
116     }
117 
118 }
