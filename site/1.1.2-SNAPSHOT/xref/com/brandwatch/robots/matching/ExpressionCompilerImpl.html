36  import com.google.common.base.Function;
37  import com.google.common.base.Optional;
38  
39  import javax.annotation.Nonnull;
40  import javax.annotation.Nullable;
41  import java.util.regex.Pattern;
42  
43  import static com.google.common.base.Preconditions.checkNotNull;
44  import static java.util.regex.Pattern.quote;
45  
46  class ExpressionCompilerImpl implements ExpressionCompiler {
47  
48      private static final String RE_LITERAL_PREFIX = "(?>";
49      private static final String RE_LITERAL_SUFFIX = ")";
50      private static final String RE_MATCH_ONE_OR_MORE = ".*?";
51      private static final String RE_MATCH_START = "^";
52      private static final String RE_MATCH_END = "$";
53  
54      private static final char EXP_MATCH_START = '^';
55      private static final char EXP_MATCH_END = '$';
56      private static final char EXP_MATCH_ONE_OR_MORE = '*';
57  
58      private final boolean caseSensitivity;
59      private final boolean leftBoundaryMatching;
60      private final Optional<Function<String, String>> expressionPreprocessor;
61  
62      ExpressionCompilerImpl(@Nonnull ExpressionCompilerBuilder builder) {
63          this.caseSensitivity = builder.isCaseSensitivity();
64          this.leftBoundaryMatching = builder.isLeftBoundaryMatching();
65          this.expressionPreprocessor = builder.getExpressionPreprocessor();
66      }
67  
68      @Nullable
69      @Override
70      public Matcher<String> compile(String expression) {
71          checkNotNull(expression, "expression");
72  
73          if (expressionPreprocessor.isPresent()) {
74              expression = expressionPreprocessor.get().apply(expression);
75          }
76  
77          if (expression.isEmpty()) {
78              return new EverythingMatcher<String>();
79          } else {
80              return new ExpressionMatcher(compileWildcardExpressionToRegex(expression), getSpecificity(expression));
81          }
82      }
83  
84      private double getSpecificity(String expression) {
85          double specificity = 0;
86          for (int i = 0; i < expression.length(); i++) {
87              final char c = expression.charAt(i);
88              if (!(c == '*' || c == '^' || c == '$')) {
89                  ++specificity;
90              }
91          }
92          return specificity;
93      }
94  
95      @Nonnull
96      private Pattern compileWildcardExpressionToRegex(@Nonnull String expression) {
97          final StringBuilder regex = new StringBuilder();
98  
99          final int len = expression.length();
100         int start = 0;
101         int end;
102 
103         if (len > 0 && expression.charAt(0) == EXP_MATCH_START) {
104             regex.append(RE_MATCH_START);
105             start++;
106         } else if (leftBoundaryMatching) {
107             regex.append(RE_MATCH_START);
108         } else {
109             regex.append(RE_MATCH_ONE_OR_MORE);
110         }
111 
112         while ((end = expression.indexOf(EXP_MATCH_ONE_OR_MORE, start)) != -1) {
113             regex.append(RE_LITERAL_PREFIX)
114                     .append(quote(expression.substring(start, end)))
115                     .append(RE_LITERAL_SUFFIX)
116                     .append(RE_MATCH_ONE_OR_MORE);
117             start = end + 1;
118         }
119 
120         if (len > 0 && expression.charAt(len - 1) == EXP_MATCH_END) {
121             regex.append(RE_LITERAL_PREFIX)
122                     .append(quote(expression.substring(start, len - 1)))
123                     .append(RE_LITERAL_SUFFIX)
124                     .append(RE_MATCH_END);
125         } else {
126             regex.append(RE_LITERAL_PREFIX)
127                     .append(quote(expression.substring(start, len)))
128                     .append(RE_LITERAL_SUFFIX)
129                     .append(RE_MATCH_ONE_OR_MORE);
130         }
131 
132         int flags = 0;
133         if (!caseSensitivity) {
134             flags |= Pattern.CASE_INSENSITIVE;
135         }
136 
137         return Pattern.compile(regex.toString(), flags);
138     }
